class Solution {
    public int coinChange(int[] coins, int amount) {
        int n=coins.length;
        Arrays.sort(coins,Comparator.reverseOrder());
        int count=0;
        for(int i=0;i<n;i++){
            if(amount>coins[i]){
                while(amount>coins[i]){
                    amount=amount-coins[i];
                    count++;
                }
            }
        }
        return count;
    }
}
