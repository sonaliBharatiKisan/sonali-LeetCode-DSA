class Solution {
    public int GetNth(Node head, int index) {
        // Code here
        Node temp=head;
        int count=1;
        while(temp!=null){
            if(count==index){
            return temp.data;
        }
        temp=temp.next;
        count++;}
       return -1;
    }
}
