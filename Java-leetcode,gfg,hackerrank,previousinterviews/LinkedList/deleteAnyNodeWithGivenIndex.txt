/* Linklist node structure
class Node
{
    int data;
    Node next;

    Node(int d)
    {
        data = d;
        next = null;
    }
}*/
class Solution {
    Node deleteNode(Node head, int x) {
        // code here
        Node temp=head;
        int count=1;
        if(x==1){
             return head.next;
                    }
    while(temp!=null && temp.next!=null){
        if(count==x-1){
            temp.next=temp.next.next;
        }
        temp=temp.next;
        count++;
    }
       return head; 
    }
}
