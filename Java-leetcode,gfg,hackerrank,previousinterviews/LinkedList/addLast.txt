class Solution {
    public Node insertAtEnd(Node head, int x) {
        // code here
        Node temp=null;
        Node newNode=new Node(x);
        if(head==null){
            return newNode;
        }
        temp=head;
        while(temp.next!=null){
            temp=temp.next;
        }
        temp.next=newNode;
        return head;
    }
}
