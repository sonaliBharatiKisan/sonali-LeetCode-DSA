class Solution {
    public long subArrayRanges(int[] nums) {
        int n=nums.length;
        int sum=0;
        for(int i=0;i<n;i++){
            int large=nums[i];
            int small=nums[i];
            for(int j=i+1;j<n;j++){
                large=Math.max(large,nums[j]);
                small=Math.min(small,nums[j]);
                sum=sum+(large-small);
            }
        }
        return sum;
    }
}
