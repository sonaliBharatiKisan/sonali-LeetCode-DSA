class Solution {
    public int sumSubarrayMins(int[] arr) {
        
        int n=arr.length;
        int mini=0;
        int sum=0;
        for(int i=0;i<n;i++){
            mini=arr[i];
            for(int j=i;j<n;j++){
                mini=Math.min(mini,arr[j]);
                sum=sum+mini;
            }
        }
        return sum;
    }
}
